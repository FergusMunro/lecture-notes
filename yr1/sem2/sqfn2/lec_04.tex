\section{Continuity}

\begin{definition}
    Let $D \subseteq \mathbb{R}$, $c \in D$ and $f : D \rightarrow \mathbb{R}$. We say $f$ is continuous at $c$ if
    \[\forall \epsilon > 0 \quad \exists \delta > 0 \quad \forall x \in D, \quad |x - c | < \delta\]
    \[|f(x) - f(c) | < \epsilon.\]

    We say $f$ is {\bf continuous} on $D$ if this holds true at every $c \in D$.
\end{definition}

\begin{theorem}
    Let $D \subseteq \mathbb{R}$, $f : D \rightarrow \mathbb{R}$ and $c$ be a cluster point of $D$.

    Then $f$ is continuous at $c$ if and only if
    \[\lim_{x \rightarrow c} f(x) = f(c).\]
\end{theorem}
\begin{proof}
    Suppose $f$ is continuous at $c$. Fix $\epsilon > 0$ and obtain $\delta$ from continuity of $f$. Then for all $x$ with $0 < |x - c| < \delta$, we have
    \[|f(x) - f(c)| < \epsilon.\]
    Thus $f$ converges to $f(c)$ at $\epsilon$. Now suppose $\lim_{x \rightarrow c}f(x) = f(c)$. Fix $\epsilon > 0$ and obtain $\delta$ from the limit of $f$. Then for all $0 < | x - c | < \delta$,
    \[|f(x) - f(c) | < \epsilon.\]
    For the case where $|x - c| = 0$, $x = c$ and $|f(c) - f(c)| = 0 < \delta$, so $f$ is continuous.
\end{proof}

\begin{remark}
  The relation between continuity at $c$ and $\lim_{x\rightarrow c} f(x) = f(c)$ does not hold if $c$ is not a cluster point. However, functions are always continuous at points that are not cluster points of the domain.
\end{remark}

\begin{theorem} (Algebra of continuous functions)

  Let $D \subseteq \mathbb{R}$, $c \in D$ and $\alpha \in \mathbb{R}$. Let $f, g : D \rightarrow \mathbb{R}$ be continuous at $c$, then
  \begin{itemize}
    \item $f + g$
    \item $\alpha f$
    \item $fg$
  \end{itemize}
  are all continuous at $c$.

  If $g(c) \not = 0$, then $\frac{f}{g}$ is also continuous at $c$.
    
\end{theorem}
\begin{proof}
    Use the algebra of limits.
\end{proof}

\subsection{Composition of Continuous Functions}

Given two sets $D, E \subseteq \mathbb{R}$, $f : D \rightarrow \mathbb{R}$ and $g : E \rightarrow D$, we can define the composite function \[(f \circ g)(x) := f(g(x)).\]

\begin{theorem} (Continuity of Continuous Functions)

  Let $D, E \subseteq \mathbb{R}$, and $c \in E$. Let $f : D \rightarrow \mathbb{R}$ and $g : E \rightarrow D$.

  If $g$ is continuous at $c$ and $f$ is continuous at $g(c)$, then $f \circ g$ is continuous at $c$.
    
\end{theorem}
\begin{proof}
    Let $\epsilon > 0$. Because $f$ is continuous at $g(c)$, there exists $\rho > 0$ such that
    \[|y - g(c)| < \rho \implies |f(y) - f(g(c))| < \epsilon.\]
    Then since $g$ is continuous at $c$ we can find a $\delta$ such that
    \[|x - c| < \delta \implies |g(x) - g(c)| < \rho.\]
\end{proof}

\subsection{Continuity of Selected Functions}

\begin{prop}
  The following functions are all continuous.

  \begin{itemize}
    \item $f(x) = \alpha$ for $\alpha \in \mathbb{R}$.
    \item $f(x) = x$
    \item $f(x) = \sum_{ i = n}^{\infty} a_n x^n$
    \item $f(x) = \exp(x)$
    \item $f(x) = \sinh(x)$
    \item $f(x) = \cosh(x)$
    \item $f(x) = \sin(x)$
    \item $f(x) = \cos(x)$
    \item $f(x) = \sec(x), g(x) = \tan(x)$ where $f, g : \mathbb{R} \setminus \{ \frac{\pi}{2} + n \pi | n \in \mathbb{Z}\} \rightarrow \mathbb{R}$
    \item $f(x) = \csc(x), g(x) = \cot(x)$ where $f, g : \mathbb{R} \setminus \{ n \pi | n \in \mathbb{Z}\} \rightarrow \mathbb{R}$

  \end{itemize}

\end{prop}
\begin{proof} $ $
  \begin{itemize}
    \item Constant and identity functions are trivial, and from these polynomials can be built.
    \item For $f(x) = \exp(x)$, we first prove $\exp(x)$ is continuous at $0$. Let $\epsilon > 0$ and
      \begin{align*}
        |\exp(x) - \exp(0)| &= |\exp(x) - 1| \\
        &= |1 + \sum_{ i = 1}^{\infty} \frac{x^n}{n!} - 1|
        &= \leq \gamma |x| \quad \text{(non-examinable)}
      .\end{align*}
      Thus if $|x| < \delta = \frac{\epsilon}{\gamma}$ then $\lim_{x \rightarrow 0} \exp(x) = 1$.

      \noindent Now for an arbitrary $c$, then
      \[\exp(x) = \exp(x - c) \exp(c)\]
      so the first term converges to $0$ which is continuous, and the second is constant thus $\exp$ is continuous on $\mathbb{R}$. This also gives continuity for $\sinh$ and $\cosh$.
    \item The same argument can prove that $\cos$ and $\sin$ are continuous at $0$. Then for any $c$,
      \begin{align*}
        \cos(x) &= \cos(x - c + c) = \cos(x - c)\cos(c) - \sin(x -c) \sin c \\
        \sin(x) &= \sin(x - c + c) = \sin(x - c) \cos (c) - \cos (x - c) \sin(c)
      \end{align*}
      which are both continuous also by the same argument. This also gives continuity for $\tan, \sec$ etc.
  \end{itemize}
\end{proof}

\subsection{The Intermediatve Value Theorem}

Let $D \subseteq \mathbb{R}$, $\alpha \in \mathbb{R}$ and $f: D \rightarrow \mathbb{R}$. Solving the equation $f(x) = \alpha$ can rarely be done explicitly. However, if $f$ is continuous on $D = [a, b]$, and if $\alpha$ is between $f(a)$ and $f(b)$, it is plausible that $f(x) = \alpha$ has a solution in $D$.

\begin{theorem}
    Let $a, b \in \mathbb{R}$ with $a < b$. Suppose
    \begin{itemize}
      \item $f:[a,b] \rightarrow \mathbb{R}$ is continuous
      \item $f(a) \leq \alpha \leq f(b)$ or $f(b) \leq \alpha \leq f(a)$.
    \end{itemize}

    Then there exists $c \in [a, b]$ such that
    \[f(c) = \alpha.\]
\end{theorem}
\begin{proof}
  We want to construct sequences $(x_n)_{n \in \mathbb{N}}$ $(y_n)_{n \in \mathbb{N}}$ that both converge to $c$ with $f(x_n) \leq \alpha$ and $f(y_n) \geq \alpha$. Then 
  \[f(c) = \lim_{n \rightarrow \infty} f(x_n) \leq \alpha \leq \lim_{n \rightarrow \infty} f(y_n) = f(c).\]
  We construct these sequences by setting $x_0 = a$ and $y_0 = b$. Then we calculate $z_0$, the midpoint, and if $f(z_0) \leq \alpha$, we set $x_1 = x_0$ and $y_1 = y_0$, and otherwise do it the other way around. This can be generalized for any $n$, giving a sequence with the desired properties.
\end{proof}
