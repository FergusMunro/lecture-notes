\section{Inner Product Spaces}

\subsection{Inner Products}

Recall that the dot (or scalar) product on $\mathbb{R}^n$ is the map
\begin{align*}
  \mathbb{R}^n \times \mathbb{R}^n &\rightarrow \mathbb{R} \\
  x \cdot y &= \sum_{ i = 1}^{n} x_i y_i.
.\end{align*}

Then we can define the length of $x$ to be $|x| = x \cdot x$, and the angle between two vectors with $x \cdot y = |x||y| \cos \theta$.

We also have the dot product on $\mathbb{C}^n$, defined as
\[x \cdot y = \sum_{ i = 1}^{n} \bar{x}_i y_i.\]
Note that $x \dot x = \sum_{ i = 1}^{n} \bar{x}_i x_i = \sum_{ i = 1}^{n} |x_i|^2$.

\begin{definition}
  Let $V$ be a vector space over $\mathbb{F}$ which we assume to be either $\mathbb{R}$ or $\mathbb{C}$. An {\bf inner product} on $V$ is a map $V \times V \rightarrow \mathbb{F} : (v, w) \mapsto \langle v, w \rangle$ which is
  \begin{itemize}
    \item {\bf congugate symmetric:} $\langle w, v \rangle = \overline{\langle v, w \rangle}$
  \item {\bf linear in the second argument:} 
    \[\langle v, \lambda u + \mu w \rangle = \lambda \langle v, u \rangle + \mu \langle v, w \rangle\]
\item {\bf positive definite:} For all $v \in V, \langle v, v \rangle \geq 0$ with equality if and only if $v = 0$
  \end{itemize}
  A vector space with a fixed inner product is called an {\bf inner product space}.
\end{definition}
$ $ \\
An inner product space comes with an inner product already defined, denoted $\langle \cdot, \cdot \rangle$.

\begin{remark} $\langle v, v \rangle = \overline{\langle v, v \rangle}$ so the inner product of a vector on itself is always real.
    \begin{itemize}
      \item Suppose $\mathbb{F} = \mathbb{R}$. Then congugate symmetry is just symmetry, and so $\langle v, w \rangle = \langle w, v \rangle$. Furthermore this means the inner product is lienar in both arguments.
      \item Now suppose $\mathbb{F} = \mathbb{C}$. Then
        \[\langle \lambda u + \mu v, w \rangle = \bar{\lambda} \langle u, w \rangle + \bar{\mu} \langle v, w \rangle.\]
    \end{itemize}
\end{remark}


\begin{definition}
    Let $v$ be an inner product space over $\mathbb{R}$.
    \begin{itemize}
      \item The {\bf norm} of a vector $v$ is $\norm{v} = \sqrt{\langle v, v \rangle} \geq 0$.

        If $\norm{v} = 1$ then we say $v$ is {\bf unit-length}, and $\frac{v}{ \norm{v}}$ is the {\bf normalisation} of $v$.
      \item Let $v, w \in V$. If $ \langle v , w \rangle = 0$, then $v$ is {\bf orthogonal} to $w$, written $v \bot w$.
    \end{itemize}
\end{definition}

\subsection{Cauchy-Schwarz Inequality}

\begin{theorem}
  Let $V$ be an inner product space. For $v, w \in V$,
  \[| \langle v, w \rangle| \leq \norm{v} \norm{w}\]
  with equality if and only if $v$ and $w$ are dependent.
\end{theorem}
\begin{proof}
    If $v = 0$ both terms equal 0, otherwise let $w = \lambda v + u$, with $v \bot u$. Then
    \begin{align*}
      \langle v, \lambda v + u \rangle &= \lambda \langle v, v \rangle + \langle v, u \rangle \\
                                       &= \lambda \norm{v}^2
    \end{align*}
    Thus $\lambda = \frac{\langle v, w \rangle}{ \norm{v}^2}$. Then using the positive definite property gives
    \begin{align*}
      0 \leq \norm{u}^2 &= \langle w - \lambda v, u \rangle \\
                        &= \langle w, u \rangle - \lambda \langle v, u \rangle \\
                        &= \langle w, w - \lambda v \rangle \\
                        &= \norm{w}^2 - \lambda \langle w, v \rangle \\
                        &= \norm{w}^2 - \frac{ \langle v, w \rangle \langle w, v \rangle}{ \norm{v}^2} \\
                      |\langle v, w \rangle|^2  &\leq \norm{v}^2 \norm{w}^2
    \end{align*}
    and we can take the square root for the final result. Then if there is equality, $\norm{u}^2 = 0$ so $w = \lambda v$ thus they are dependent.
\end{proof}

\begin{corollary} (Triangle Inequality)
  For $v, w \in V$ we have
  \[\norm{v + w} \leq \norm{v} + \norm{w}\]
  with equality if and only if $v$ and $w$ are dependent.
\end{corollary}
\begin{proof}
  By expanding out $\langle v + w, v+w \rangle$ it can be shown
    \begin{align*}
      \norm{v + w}^2 &= \langle v + w, v + w \rangle^2 \\
       &= \norm{v}^2 + \norm{w}^2 + \langle v, w \rangle + \overline{ \langle v, w \rangle} \\
       &= \norm{v}^2 + \norm{w}^2 + 2 Re (\langle v, w \rangle)
    .\end{align*}
    Then $Re( \langle v, w \rangle) \leq | \langle v, w \rangle| \leq \norm{v}\norm{w}$ so
    \begin{align*}
      \norm{v + w}^2 &\leq \norm{v}^2 + 2 \norm{v} \norm{w} \norm{w}^2 \\
                     &= (\norm{v} + \norm{w})^2
    \end{align*}
    and the final result can be obtained by taking the square root.
\end{proof}

\subsection{Orthogonality}

\begin{definition}
  A list $u_1,...,u_k$ in an inner product space $V$ is {\bf orthonormal} if for $1 \leq i, j \leq k$
  \[\langle u_i, u_j \rangle = \delta_{ij} = \left\{\begin{array}{lr}
  1 & i = j\\
  0 & i \not = j\\
  \end{array} \right..\]
\end{definition}

\begin{remark}
    A list is orthonormal if every $u_i$ has unit length and distinct $u_i,u_j$ are orthogonal.
\end{remark}

\begin{lemma}
  For an orthonormal basis $u_1,...,u_n$ in an inner product space $V$, for a vector $v \in V$ we have
    \[v = \sum_{i = 1}^n \langle u_i, v \rangle u_i.\]
\end{lemma}
\begin{proof}
  Since $u_1, ..., u_n$ are a basis $v = \sum_{ i = 1}^{n} \lambda_i u_i$. Furthermore
  \begin{align*}
    \langle u_i, v \rangle &= \sum_{ j = 1}^{n} \lambda_j \langle u_i, u_j \rangle \\
                           &= 0 + ... + \lambda_i + ... + 0
  .\end{align*}
  Thus if every $\lambda_i = \langle u_i, v \rangle$ then the final result is obtained.
\end{proof}

\begin{lemma}
    Any orthonormal list is linearly independent.
\end{lemma}
\begin{proof}
    If $v = 0$, then $\langle u_i, v \rangle = 0$ so all $\lambda_i = 0$ so the list must be independent.
\end{proof}

\begin{prop}
    If $u_1,...,u_n$ is an orthonormal basis for $V$ and 
    \begin{align*}
      v &= \sum_{ i = 1}^{n} x_i u_i \\
      w &= \sum_{ i = 1}^{n} y_i u_i
    \end{align*}
    then \[\langle v, w \rangle = \sum_{ i = 1}^{n} \bar{x}_i y_i = x \cdot y\]
\end{prop}
\begin{proof}
    Expand using sesquilinearity.
    \begin{align*}
      \langle v, w \rangle &= \langle \sum_{ i = 1}^{n} x_i u_i, \sum_{ j = 1}^{n} y_j u_j \rangle \\
                           &= \sum_{ i = 1}^{n} \sum_{ j = 1}^{n} \bar{x}_i y_i \langle u_i, u_j \rangle \\
                           &= \sum_{ i = 1}^{n} \bar{x}_i y_i
    .\end{align*}
\end{proof}

\begin{theorem} (Gram-Schmidt Orthogonalisation)
  Let $v_1, ..., v_n$ be an independent list in an inner product space $V$. Then there exists an orthonormal list $u_1,...,u_n$ such that for all $1 \leq k \leq u$ we have
  \[span \ \{v_1,...,v_k\} = span \ \{u_1,...,u_k\}.\]
  We construct $u_1,...,u_k$ inductively:
  \begin{itemize}
    \item For $k = 1$ we set $w_1 = v_1$, and $u_1 = \frac{w_1}{\norm{w_1}}$
    \item For $k > 1$, suppose we have defined $u_1,...,u_k$ already and set 
  \begin{align*}
    w_k &= v_k - \sum_{ j = 1}^{k - 1} \langle u_j, v_k \rangle u_j \\
    &= v_k - \sum_{ j = 1}^{k-1} \frac{\langle w_j, v_k \rangle}{\norm{w_j}^2} w_j
  .\end{align*}
  and $u_k = \frac{w_k}{\norm{w_k}}$.

  \end{itemize}
\end{theorem}
\begin{proof}
  Using induction on $k$, we see the base case is trivial. Then given $u_1,...,u_{k-1}$ orthogonal with span equal to that of $v_1,...,v_{n-1}$. Then
  \begin{align*}
    \Span\{ u_1,...,u_k\} &= \Span \{v_1,...,u_k\} = \Span \{v_1,..., w_k\} \\
     &= \Span \{v_1,..., v_k - \sum_{ i = 1}^{n} \langle u_j, v_k \rangle u_j\} \\
     &= \Span \{v_1,...,v_n\}
  .\end{align*}
  Next, we need to show $w_k$ is orthogonal to all $u_i$, $i < k$. Then
  \begin{align*}
    \langle u_i, w_k \rangle &= \langle u_i, v_k - \sum_{ i = 1}^{k-1} \langle u_i, v_k \rangle u_j \rangle \\
    f &= \langle u_i, v_k \rangle - \sum_{ j = 1}^{k-1} \langle u_j, v_k \rangle \langle u_i, u_j \rangle \\
    f &= \langle u_i, v_k \rangle - \langle u_i, v_k \rangle = 0
  .\end{align*}
\end{proof}

\begin{corollary}
    Any finite dimensional inner product space has an orthonormal basis.
\end{corollary}
\begin{proof}
  Apply Gram-Schmidt orthogonalisation to a basis of $V$.
\end{proof}

\begin{remark}
    In practise it is easier to start with $v_1,...,v_n$, then product an $w_1,...,w_n$ satisfying $w_1 \bot w_j (i \not = j)$ then finally set $u_i = \frac{w_i}{ \norm{w_i}}$ for every $i$ to get an orthonormal list.
\end{remark}
